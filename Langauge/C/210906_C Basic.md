``` c
    int input;
    printf("값을 입력하세요 : ");
    // & :: 값을 입력할것이다
    scanfs("%d", &input);
    printf("입력값 : %d\n", input);
    
    int one, two, three;
    printf("3 개의 정수를 입력하세요");
    scanf_s("%d %d %d", &one, %two, %three);
    printf("첫 번째 값 :: %d \n", one);
    printf("두 번째 값 :: %d \n", two);
    printf("세 번째 값 :: %d \n", three);
    

    /* 문자 (한 글자) */
    char c = 'A';
    printf("%c", c);

    /* 문자열 (한 글자 이상의 여러 글자) */
    char str[256];
    scanf_s("%s", str, sizeof(str));

    return 0;
  }
```

## 증감 연산자

``` c
  int main () {
    // ++
    int a = 10;
    printf("a: %d \n", a); // 10
    a++;
    printf("a: %d \n", a); // 11
    

    int b = 20;
    printf("b: %d \n", ++b); // 21 :: 이 코드를 실행하기 전 ++
    printf("b: %d \n", b); // 21
    printf("b: %d \n", b++); // 21 :: 이 코드를 실행한 후 ++
    printf("b: %d \n", b); // 22

    return 0;
  }
```

## 반복문

``` c
int main () {
  // for
  for (int i = 1; i <= 10; i++) {
    printf("i: %d \n", i);  // 1 ~ 10
  }

  // while
  int i = 1;
  while (i <= 10) {
      printf("Hello World, %d \n", i++);
  }

  // do while
  int i = 1;
  do {
      printf("HelloWorld %d \n", i++);
  } while (i <= 10);

  /* Break */
  for (int i = 1; i <= 30; i++) {
    if (i >= 6) {
      printf("나머지 학생은 GO HOME \n");
      break; // 탈출
    }
    printf("%d 번 학생 발표 \n", i);
  }

  /* Continue */
  for (int i = 1; i <= 30; i++) {
    if (i >= 6 && i <= 10) {
      if (i == 7) {
        printf("%d 번 학생 결석... \n", i);
        continue; // 이 조건을 제외하고 계속
      }
      printf("%d 번 학생 발표 \n", i);
    }
  }
  
  return 0;
}
```

``` c
  }
```

## 문자 vs 문자열

``` c
int main () {
  
  char c = 'A';
  printf("%c \n", c); // A
  
  // 문자열 끝에는 '끝' 을 의미하는 NULL 문자 \0 이 포함
  char str[6] = "coding"; // [c], [o], [d], [i], [n], [g], [\0]
  printf("%s \n", str); // coding

  char str[] = "coding";
  printf("%s \n", str); // coding
  printf("%d \n", sizeof(str)); // 7  :: 6 + NULL 문자 포함
  
  char str[] = "coding";
  for (int i = 0; i < sizeof(str); i++) {
      printf("%c \n", str[i]); // coding" "
  }

  // 영어 1 글자 : 1byte
  // 한글 1 글자 : 2byte
  char kor[] = "나도 코딩";
  printf("%s \n", kor);
  printf("%d \n", sizeof(kor)); // 14


  char c_array[7] = { 'c', 'o', 'd', 'i', 'n', 'g', '\0'  };
  printf("%s \n", c_array); // coding" "


  char c_array[10] = { 'c', 'o', 'd', 'i', 'n', 'g' };
  for (int i = 0; i < sizeof(c_array); i++) {
      printf("%c \n", c_array[i]); // coding" " " " " " " "
      printf("%d \n", c_array[i]); // 99 111 100 105 110 103 0 0 0 0  :: ASCII 코드
  }

  
  return 0;
}

```

## ASCII Code 문자

``` c
#include <stdio.h>

int main() {
  // 참고 :: ASCII 코드
  // 미국 표준 협회 ANSI 에서 제시한 표준 코드 체계
  // 7 bit, 총 128 개 코드
  // a : 97 (문자 a 의 아스키 코드 정수값)
  // A : 65
  // 0: 48
  
  printf("%c \n", 'a'); // a
  printf("%d \n", 'a'); // 97
  
  printf("%c \n", 'b'); // b
  printf("%d \n", 'b'); // 98
  
  printf("%c \n", '\0'); // " "
  printf("%d \n", '\0'); // 0
  
  printf("%c \n", '0'); // 0
  printf("%d \n", '0'); // 48
  
  printf("%c \n", '1'); // 1
  printf("%d \n", '1'); // 49
  
  // 참고 :: 0 ~ 127 사이의 ASCII 코드 
  for (int i = 0; i < 128; i++) {
      printf("ASCII Code Int %d :: %c\n", i, i);
      // ASCII Code 전체 출력
  }
  
  return 0;
}
```

``` C
// Online C compiler to run C program online
#include <stdio.h>

  // [철수] : 101 호

  // 첫 번째 미션 : 아파트에 각 집에 방문하여 문에 적힌 암호 확인
  int *missionM = &c;  // 포인터 변수

  printf("미션맨이 방문하는 주소: %d, 암호: %d \n", missionM, *missionM);
  // 방문하는 주소 : 320960596, 암호: 1
  // 철수네와 동일
  
  missionM = &y;
  printf("미션맨이 방문하는 주소: %d, 암호: %d \n", missionM, *missionM);
  // 방문하는 주소 : 320960600, 암호: 2
  // 영희네와 동일
  
  missionM = &m;

  // 방문하는 주소 : 320960604, 암호: 3
  // 민수네와 동일
  
  // 두 번째 미션 : 각 암호에 3을 곱하기

  *missionM = *missionM * 3;
  printf("미션맨이 암호를 바꾼곳 주소: %d, 암호: %d \n", missionM, *missionM);
```

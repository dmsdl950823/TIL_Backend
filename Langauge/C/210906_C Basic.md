# C Basic

``` c
  // #include <함수들을 모아놓은 헤더 파일>
  #include <stdio.h>

  int main (void) {
    /* 변수 */
    // 자료형 변수명 = 값
    int age = 12;
    
    // %d 정수형 값 출력 + ',' 뒤의 값 순서대로 출력
    printf("%d \n", age); // 12

    age = 13;
    printf("%d \n", age); // 13

    /* 실수형 변수 */
    float f = 46.5;
    printf("%f \n", f); // 46.500000
    printf("%.2f \n", f); // 46.50
    
    double d = 4.428;
    printf("%f \n", d); // 4.428000
    printf("%.2lf \n", d); // 4.3

    /* 상수 */
    const int YEAR = 2000;
    printf("태어난 년도 : %d \n", YEAR);
    // YEAR = 2001; // Error!

      
    /* scanf :: 키보드 입력을 받아서 저장 */
    int input;
    printf("값을 입력하세요 : ");
    // & :: 값을 입력할것이다
    scanfs("%d", &input);
    printf("입력값 : %d\n", input);
    
    int one, two, three;
    printf("3 개의 정수를 입력하세요");
    scanf_s("%d %d %d", &one, %two, %three);
    printf("첫 번째 값 :: %d \n", one);
    printf("두 번째 값 :: %d \n", two);
    printf("세 번째 값 :: %d \n", three);
    

    /* 문자 (한 글자) */
    char c = 'A';
    printf("%c", c);

    /* 문자열 (한 글자 이상의 여러 글자) */
    char str[256];
    scanf_s("%s", str, sizeof(str));
    printf("%s \n", str);

    return 0;
  }
```
